{
    int idx;
    int jdx;
    int temp;

    for (idx = 0; idx < n - 1; idx++)
    {
        for (jdx = 0; jdx < (n - idx) - 1; jdx++)
        {
            if (arr[jdx] > arr[jdx + 1])
            {
                temp = arr[jdx];
                arr[jdx] = arr[jdx + 1];

            }
        }
    }
}
